71  static inline int GetBytes(const uint8_t* pData, int dataSize, void* pBuf, int size, int startPos)
72  {
73  int endPos = startPos + size;
74  if (startPos >= 0 && endPos <= dataSize)
75  {
76  memcpy(pBuf, pData + startPos, size);
77  return endPos;
78  }
79  return -1;
80  }
81 
88  static inline int GetStr(const uint8_t* pData, int dataSize, WDL_String& str, int startPos)
89  {
90  int len;
91  int strStartPos = GetBytes(pData, dataSize, &len, sizeof(len), startPos);
92  if (strStartPos >= 0)
93  {
94  int strEndPos = strStartPos + len;
95  if (strEndPos <= dataSize)
96  {
97  if (len > 0)
98  str.Set((char*) (pData + strStartPos), len);
99  else
100  str.Set("");
101  }
102  return strEndPos;
103  }
104  return -1;
105  }
