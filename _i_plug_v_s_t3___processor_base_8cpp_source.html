66 void IPlugVST3ProcessorBase::ProcessMidiIn(IEventList* eventList, IPlugQueue<IMidiMsg>& editorQueue, IPlugQueue<IMidiMsg>& processorQueue)
67 {
68  IMidiMsg msg;
69 
70  // Process events.. only midi note on and note off?
71 
72  if (eventList)
73  {
74  int32 numEvent = eventList->getEventCount();
75  for (int32 i=0; i<numEvent; i++)
76  {
77  Event event;
78  if (eventList->getEvent(i, event) == kResultOk)
79  {
80  switch (event.type)
81  {
82  case Event::kNoteOnEvent:
83  {
84  msg.MakeNoteOnMsg(event.noteOn.pitch, event.noteOn.velocity * 127, event.sampleOffset, event.noteOn.channel);
85  ProcessMidiMsg(msg);
86  processorQueue.Push(msg);
87  break;
88  }
89 
90  case Event::kNoteOffEvent:
91  {
92  msg.MakeNoteOffMsg(event.noteOff.pitch, event.sampleOffset, event.noteOff.channel);
93  ProcessMidiMsg(msg);
94  processorQueue.Push(msg);
95  break;
96  }
97  case Event::kPolyPressureEvent:
98  {
99  msg.MakePolyATMsg(event.polyPressure.pitch, event.polyPressure.pressure * 127., event.sampleOffset, event.polyPressure.channel);
100  ProcessMidiMsg(msg);
101  processorQueue.Push(msg);
102  break;
103  }
104  case Event::kDataEvent:
105  {
106  ISysEx syx = ISysEx(event.sampleOffset, event.data.bytes, event.data.size);
107  ProcessSysEx(syx);
108  //mSysexMsgsFromProcessor.Push
109  break;
110  }
111  }
112  }
113  }
114  }
115 
116  while (editorQueue.Pop(msg))
117  {
118  ProcessMidiMsg(msg);
119  }
120 }

122 void IPlugVST3ProcessorBase::ProcessMidiOut(IPlugQueue<SysExData>& sysExQueue, SysExData& sysExBuf, IEventList* outputEvents, int32 numSamples)
123 {
124  // MIDI
125  if (!mMidiOutputQueue.Empty() && outputEvents)
126  {
127  Event toAdd = {0};
128  IMidiMsg msg;
129 
130  while (!mMidiOutputQueue.Empty())
131  {
132  IMidiMsg& msg = mMidiOutputQueue.Peek();
133 
134  if (msg.StatusMsg() == IMidiMsg::kNoteOn)
135  {
136  toAdd.type = Event::kNoteOnEvent;
137  toAdd.noteOn.channel = msg.Channel();
138  toAdd.noteOn.pitch = msg.NoteNumber();
139  toAdd.noteOn.tuning = 0.;
140  toAdd.noteOn.velocity = (float) msg.Velocity() * (1.f / 127.f);
141  toAdd.noteOn.length = -1;
142  toAdd.noteOn.noteId = -1; // TODO ?
143  toAdd.sampleOffset = msg.mOffset;
144  outputEvents->addEvent(toAdd);
145  }
146  else if (msg.StatusMsg() == IMidiMsg::kNoteOff)
147  {
148  toAdd.type = Event::kNoteOffEvent;
149  toAdd.noteOff.channel = msg.Channel();
150  toAdd.noteOff.pitch = msg.NoteNumber();
151  toAdd.noteOff.velocity = (float) msg.Velocity() * (1.f / 127.f);
152  toAdd.noteOff.noteId = -1; // TODO ?
153  toAdd.sampleOffset = msg.mOffset;
154  outputEvents->addEvent(toAdd);
155  }
156  else if (msg.StatusMsg() == IMidiMsg::kPolyAftertouch)
157  {
158  toAdd.type = Event::kPolyPressureEvent;
159  toAdd.polyPressure.channel = msg.Channel();
160  toAdd.polyPressure.pitch = msg.NoteNumber();
161  toAdd.polyPressure.pressure = (float) msg.PolyAfterTouch() * (1.f / 127.f);
162  toAdd.polyPressure.noteId = -1; // TODO ?
163  toAdd.sampleOffset = msg.mOffset;
164  outputEvents->addEvent(toAdd);
165  }
166  else if (msg.StatusMsg() == IMidiMsg::kControlChange)
167  {
168  toAdd.type = Event::kLegacyMIDICCOutEvent;
169  toAdd.midiCCOut.channel = msg.Channel();
170  toAdd.midiCCOut.controlNumber = msg.mData1;
171  toAdd.midiCCOut.value = msg.mData2;
172  toAdd.midiCCOut.value2 = 0;
173  }
174  else if (msg.StatusMsg() == IMidiMsg::kPitchWheel)
175  {
176  toAdd.type = Event::kLegacyMIDICCOutEvent;
177  toAdd.midiCCOut.channel = msg.Channel();
178  toAdd.midiCCOut.value = msg.mData1;
179  toAdd.midiCCOut.value = msg.mData2;
180  }
181 
182  mMidiOutputQueue.Remove();
183  // don't add any midi messages other than noteon/noteoff
184  }
185  }
186 
187  mMidiOutputQueue.Flush(numSamples);
188 
189  // Output SYSEX from the editor, which has bypassed the processors' ProcessSysEx()
190  if (sysExQueue.ElementsAvailable())
191  {
192  Event toAdd = {0};
193 
194  while (sysExQueue.Pop(sysExBuf))
195  {
196  toAdd.type = Event::kDataEvent;
197  toAdd.sampleOffset = sysExBuf.mOffset;
198  toAdd.data.type = DataEvent::kMidiSysEx;
199  toAdd.data.size = sysExBuf.mSize;
200  toAdd.data.bytes = (uint8*) sysExBuf.mData; // TODO! this is a problem if more than one message in this block!
201  outputEvents->addEvent(toAdd);
202  }
203  }
204 }

315 void IPlugVST3ProcessorBase::ProcessParameterChanges(ProcessData& data)
316 {
317  IParameterChanges* paramChanges = data.inputParameterChanges;
318 
319  if (paramChanges)
320  {
321  int32 numParamsChanged = paramChanges->getParameterCount();
322 
323  for (int32 i = 0; i < numParamsChanged; i++)
324  {
325  IParamValueQueue* paramQueue = paramChanges->getParameterData(i);
326  if (paramQueue)
327  {
328  int32 numPoints = paramQueue->getPointCount();
329  int32 offsetSamples;
330  double value;
331 
332  if (paramQueue->getPoint(numPoints - 1, offsetSamples, value) == kResultTrue)
333  {
334  int idx = paramQueue->getParameterId();
335 
336  switch (idx)
337  {
338  case kBypassParam:
339  {
340  const bool bypassed = (value > 0.5);
341 
342  if (bypassed != GetBypassed())
343  SetBypassed(bypassed);
344 
345  break;
346  }
347  default:
348  {
349  if (idx >= 0 && idx < mPlug.NParams())
350  {
351  ENTER_PARAMS_MUTEX;
352  mPlug.GetParam(idx)->SetNormalized((double)value); // TODO: In VST3 non distributed the same parameter value is also set via IPlugVST3Controller::setParamNormalized(ParamID tag, ParamValue value)
353  mPlug.OnParamChange(idx, kHost, offsetSamples);
354  LEAVE_PARAMS_MUTEX;
355  }
356  }
357  break;
358  }
359  }
360  }
361  }
362  }

365 void IPlugVST3ProcessorBase::ProcessAudio(ProcessData& data, ProcessSetup& setup, const BusList& ins, const BusList& outs)
366 {
367  int32 sampleSize = setup.symbolicSampleSize;
368 
369  if (sampleSize == kSample32 || sampleSize == kSample64)
370  {
371  if (data.numInputs)
372  {
373  if (HasSidechainInput())
374  {
375  if (IsBusActive(ins, 1)) // Sidechain is active
376  {
377  mSidechainActive = true;
378  SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), true);
379  }
380  else
381  {
382  if (mSidechainActive)
383  {
384  ZeroScratchBuffers();
385  mSidechainActive = false;
386  }
387 
388  SetChannelConnections(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), true);
389  SetChannelConnections(ERoute::kInput, data.inputs[0].numChannels, MaxNChannels(ERoute::kInput) - NSidechainChannels(), false);
390  }
391 
392  AttachBuffers(ERoute::kInput, 0, MaxNChannels(ERoute::kInput) - NSidechainChannels(), data.inputs[0], data.numSamples, sampleSize);
393  AttachBuffers(ERoute::kInput, NSidechainChannels(), MaxNChannels(ERoute::kInput) - NSidechainChannels(), data.inputs[1], data.numSamples, sampleSize);
394  }
395  else
396  {
397  SetChannelConnections(ERoute::kInput, 0, data.inputs[0].numChannels, true);
398  SetChannelConnections(ERoute::kInput, data.inputs[0].numChannels, MaxNChannels(ERoute::kInput) - data.inputs[0].numChannels, false);
399  AttachBuffers(ERoute::kInput, 0, MaxNChannels(ERoute::kInput), data.inputs[0], data.numSamples, sampleSize);
400  }
401  }
402 
403  for (int outBus = 0, chanOffset = 0; outBus < data.numOutputs; outBus++)
404  {
405  int busChannels = data.outputs[outBus].numChannels;
406  SetChannelConnections(ERoute::kOutput, chanOffset, busChannels, IsBusActive(outs, outBus));
407  SetChannelConnections(ERoute::kOutput, chanOffset + busChannels, MaxNChannels(ERoute::kOutput) - (chanOffset + busChannels), false);
408  AttachBuffers(ERoute::kOutput, chanOffset, busChannels, data.outputs[outBus], data.numSamples, sampleSize);
409  chanOffset += busChannels;
410  }
411 
412  if (GetBypassed())
413  {
414  if (sampleSize == kSample32)
415  PassThroughBuffers(0.f, data.numSamples); // single precision
416  else
417  PassThroughBuffers(0.0, data.numSamples); // double precision
418  }
419  else
420  {
421  if (sampleSize == kSample32)
422  ProcessBuffers(0.f, data.numSamples); // single precision
423  else
424  ProcessBuffers(0.0, data.numSamples); // double precision
425  }
426  }
427 }
