150 tresult PLUGIN_API IPlugVST3Processor::notify(IMessage* message)
151 {
152  if (!message)
153  return kInvalidArgument;
154 
155  const void* data = nullptr;
156  uint32 size;
157 
158  if (!strcmp(message->getMessageID(), "SMMFUI")) // midi message from UI
159  {
160  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
161  {
162  if (size == sizeof(IMidiMsg))
163  {
164  IMidiMsg msg;
165  memcpy(&msg, data, size);
166  mMidiMsgsFromEditor.Push(msg);
167  return kResultOk;
168  }
169 
170  return kResultFalse;
171  }
172  }
173  else if (!strcmp(message->getMessageID(), "SAMFUI")) // message from UI
174  {
175  int64 messageTag;
176  int64 controlTag;
177 
178  if (message->getAttributes()->getInt("MT", messageTag) == kResultOk && message->getAttributes()->getInt("CT", controlTag) == kResultOk)
179  {
180  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
181  {
182  if(OnMessage((int) messageTag, (int) controlTag, size, data))
183  {
184  return kResultOk;
185  }
186  }
187 
188  return kResultFalse;
189  }
190  }
191 
192  return AudioEffect::notify(message);
193 }
