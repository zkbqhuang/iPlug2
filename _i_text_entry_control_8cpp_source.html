193 bool ITextEntryControl::OnKeyDown(float x, float y, const IKeyPress& key)
194 {
195  if (key.C)
196  {
197  switch (key.VK)
198  {
199  case 'A':
200  {
201  CallSTB([&] {
202  mEditState.select_start = 0;
203  mEditState.select_end = mEditString.GetLength();
204  });
205  return true;
206  }
207  case 'X':
208  {
209  CopySelection();
210  CallSTB([&] {
211  stb_textedit_cut(this, &mEditState);
212  });
213  return true;
214  }
215  case 'C':
216  {
217  CopySelection();
218  return true;
219  }
220  case 'V':
221  {
222  WDL_String fromClipboard;
223  if (GetUI()->GetTextFromClipboard(fromClipboard))
224  {
225  CallSTB([&] {
226  stb_textedit_paste(this, &mEditState, fromClipboard.Get(), fromClipboard.GetLength());
227  });
228  }
229  return true;
230  }
231  case 'Z':
232  {
233  if (key.S)
234  {
235  CallSTB([&]() { stb_textedit_key(this, &mEditState, STB_TEXTEDIT_K_REDO); });
236  }
237  else
238  {
239  CallSTB([&]() { stb_textedit_key(this, &mEditState, STB_TEXTEDIT_K_UNDO); });
240  }
241  return true;
242  }
243 
244  default:
245  break;
246  }
247  }
248 
249  int stbKey;
250 
251  wdl_utf8_parsechar(key.utf8, &stbKey);
252 
253  switch (key.VK)
254  {
255  case kVK_SPACE: stbKey = ' '; break;
256  case kVK_TAB: return false;
257  case kVK_DELETE: stbKey = STB_TEXTEDIT_K_DELETE; break;
258  case kVK_BACK: stbKey = STB_TEXTEDIT_K_BACKSPACE; break;
259  case kVK_RETURN: CommitEdit(); break;
260  case kVK_ESCAPE: DismissEdit(); break;
261  default:
262  {
263  // validate input based on param type
264  const IParam* pParam = GetUI()->mInTextEntry->GetParam();
265 
266  if(pParam)
267  {
268  switch (pParam->Type())
269  {
270  case IParam::kTypeEnum:
271  case IParam::kTypeInt:
272  case IParam::kTypeBool:
273  {
274  if (key.VK >= '0' && key.VK <= '9')
275  break;
276  if (key.VK >= kVK_NUMPAD0 && key.VK <= kVK_NUMPAD9)
277  break;
278  if (stbKey == '+' || stbKey == '-')
279  break;
280  stbKey = 0;
281  break;
282  }
283  case IParam::kTypeDouble:
284  {
285  if (key.VK >= '0' && key.VK <= '9')
286  break;
287  if (key.VK >= kVK_NUMPAD0 && key.VK <= kVK_NUMPAD9)
288  break;
289  if (stbKey == '+' || stbKey == '-' || stbKey == '.')
290  break;
291  stbKey = 0;
292  break;
293  }
294  default:
295  break;
296  }
297  }
298 
299  if (stbKey == 0)
300  {
301  stbKey = (key.VK) | VIRTUAL_KEY_BIT;
302  }
303  break;
304  }
305  }
306 
307  if (key.C)
308  stbKey |= STB_TEXTEDIT_K_CONTROL;
309  if (key.A)
310  stbKey |= STB_TEXTEDIT_K_ALT;
311  if (key.S)
312  stbKey |= STB_TEXTEDIT_K_SHIFT;
313 
314  return CallSTB([&]() { stb_textedit_key(this, &mEditState, stbKey); }) ? true : false;
315 }

365 void ITextEntryControl::Layout(StbTexteditRow* row, ITextEntryControl* _this, int start_i)
366 {
367  assert (start_i == 0);
368 
369  _this->FillCharWidthCache();
370  float textWidth = 0.;
371 
372  for (int i = 0; i < _this->mCharWidths.GetSize(); i++) {
373  textWidth += _this->mCharWidths.Get()[i];
374  }
375 
376  row->num_chars = _this->mEditString.GetLength();
377  row->baseline_y_delta = 1.25;
378 
379  switch (_this->GetText().mAlign)
380  {
381  case IText::kAlignNear:
382  {
383  row->x0 = _this->GetRECT().L + 1;
384  row->x1 = row->x0 + textWidth;
385  break;
386  }
387  case IText::kAlignCenter:
388  {
389  row->x0 = roundf(_this->GetRECT().MW() - (textWidth * 0.5f));
390  row->x1 = row->x0 + textWidth;
391  break;
392  }
393  case IText::kAlignFar:
394  {
395  row->x0 = _this->GetRECT().R - textWidth;
396  row->x1 = row->x0 + textWidth;
397  }
398  default:
399  {
400  assert(true);// not implemented
401  break;
402  }
403  }
404 
405  switch (_this->GetText().mVAlign)
406  {
407  case IText::kVAlignTop:
408  {
409  row->ymin = 0;
410  break;
411  }
412 
413  case IText::kVAlignMiddle:
414  {
415  row->ymin = _this->GetRECT().H()*0.5f - _this->GetText().mSize*0.5f;
416  break;
417  }
418 
419  case IText::kVAlignBottom:
420  {
421  row->ymin = _this->GetRECT().H() - _this->GetText().mSize;
422  break;
423  }
424 
425  default:
426  {
427  assert(false); // not implemented
428  break;
429  }
430  }
431 
432  row->ymax = row->ymin + static_cast<float> (_this->GetText().mSize);
433 }

476 // the width of character 'c' should include the kerning between it and the next character,
477 // so that when adding up character widths in the cache we can get to the beginning of the visible glyph,
478 // which is important for cursor placement to look correct.
479 // stb_textedit behavior for clicking in the text field is to place the cursor in front of a character
480 // when the xpos is less than halfway into the width of the character and in front of the following character otherwise.
481 // see: https://github.com/nothings/stb/issues/6
482 float ITextEntryControl::GetCharWidth(char c, char nc)
483 {
484  IRECT bounds;
485 
486  if (nc)
487  {
488  WDL_String ncstr; ncstr.SetFormatted(1, "%c", nc);
489  WDL_String cstr; cstr.SetFormatted(1, "%c", c);
490 
491  GetUI()->MeasureText(mText, ncstr.Get(), bounds);
492  float ncWidth = bounds.W();
493  cstr.Append(&ncstr);
494  GetUI()->MeasureText(mText, cstr.Get(), bounds);
495  float tcWidth = bounds.W();
496  return tcWidth - ncWidth;
497  }
498 
499  WDL_String cstr; cstr.SetFormatted(1, "%c", c);
500  GetUI()->MeasureText(mText, cstr.Get(), bounds);
501  return bounds.W();
502 }
