32 inline LICE_pixel LiceColor(const IColor& color)
33 {
34  auto preMul = [](int color, int A) {return (color * (A + 1)) >> 8; };
35  return LICE_RGBA(preMul(color.R, color.A), preMul(color.G, color.A), preMul(color.B, color.A), color.A);
36 }

38 inline LICE_pixel LiceColor(const IColor& color, const IBlend* pBlend)
39 {
40  int alpha = std::round(color.A * BlendWeight(pBlend));
41  return LICE_RGBA(color.R, color.G, color.B, alpha);
42 }

44 inline int LiceBlendMode(const IBlend* pBlend)
45 {
46  if (!pBlend)
47  {
48  return LICE_BLIT_MODE_COPY | LICE_BLIT_USE_ALPHA;
49  }
50  switch (pBlend->mMethod)
51  {
52  case EBlendType::kBlendClobber: return LICE_BLIT_MODE_COPY;
53  case EBlendType::kBlendAdd: return LICE_BLIT_MODE_ADD | LICE_BLIT_USE_ALPHA;
54  case EBlendType::kBlendDefault:
55  default:
56  {
57  return LICE_BLIT_MODE_COPY | LICE_BLIT_USE_ALPHA;
58  }
59  }
60 }
