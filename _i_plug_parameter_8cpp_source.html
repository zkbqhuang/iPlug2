237 void IParam::GetDisplayForHost(double value, bool normalized, WDL_String& str, bool withDisplayText) const
238 {
239  if (normalized) value = FromNormalized(value);
240 
241  if (mDisplayFunction != nullptr)
242  {
243  mDisplayFunction(value, str);
244  return;
245  }
246 
247  if (withDisplayText)
248  {
249  const char* displayText = GetDisplayText(value);
250 
251  if (CStringHasContents(displayText))
252  {
253  str.Set(displayText, MAX_PARAM_DISPLAY_LEN);
254  return;
255  }
256  }
257 
258  double displayValue = value;
259 
260  if (mFlags & kFlagNegateDisplay)
261  displayValue = -displayValue;
262 
263  // Squash all zeros to positive
264  if (!displayValue) displayValue = 0.0;
265 
266  if (mDisplayPrecision == 0)
267  {
268  str.SetFormatted(MAX_PARAM_DISPLAY_LEN, "%d", static_cast<int>(round(displayValue)));
269  }
270  else if ((mFlags & kFlagSignDisplay) && displayValue)
271  {
272  char fmt[16];
273  sprintf(fmt, "%%+.%df", mDisplayPrecision);
274  str.SetFormatted(MAX_PARAM_DISPLAY_LEN, fmt, displayValue);
275  }
276  else
277  {
278  str.SetFormatted(MAX_PARAM_DISPLAY_LEN, "%.*f", mDisplayPrecision, displayValue);
279  }
280 }
