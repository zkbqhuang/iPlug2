19 class alpha_span_generator : public SpanGeneratorType
20 {
21 public:
22  alpha_span_generator(typename SpanGeneratorType::source_type& source, typename SpanGeneratorType::interpolator_type& interpolator, agg::cover_type a)
23  : SpanGeneratorType(source, interpolator), alpha(a) {}
24 
25  void generate(typename SpanGeneratorType::color_type* span, int x, int y, unsigned len)
26  {
27  SpanGeneratorType::generate(span, x, y, len);
28 
29  if (alpha != 255)
30  {
31  for (unsigned i = 0; i < len; i++, span++)
32  span->a = (span->a * alpha + SpanGeneratorType::base_mask) >> SpanGeneratorType::base_shift;
33  }
34  }
35 
36 private:
37  agg::cover_type alpha;
38 };

159  void BlendFrom(agg::rendering_buffer& renBuf, const IRECT& bounds, int srcX, int srcY, agg::comp_op_e op, agg::cover_type cover, bool preMultiplied)
160  {
161  // N.B. blend_from/rect_i is inclusive, hence -1 on each dimension here
162  agg::rect_i r(srcX, srcY, srcX + std::round(bounds.W()) - 1, srcY + std::round(bounds.H()) - 1);
163  int x = std::round(bounds.L) - srcX;
164  int y = std::round(bounds.T) - srcY;
165 
166  if (preMultiplied)
167  {
168  mPixfPre.comp_op(op);
169  mRenBasePre.blend_from(PixfmtType(renBuf), &r, x, y, cover);
170  }
171  else
172  {
173  mPixf.comp_op(op);
174  mRenBase.blend_from(PixfmtType(renBuf), &r, x, y, cover);
175  }
176  }

187  template <typename VertexSourceType>
188  void SetPath(VertexSourceType& path)
189  {
190  // Clip
191  IRECT clip = mGraphics.mClipRECT.Empty() ? mGraphics.GetBounds() : mGraphics.mClipRECT;
192  clip.Translate(mGraphics.XTranslate(), mGraphics.YTranslate());
193  clip.Scale(mGraphics.GetBackingPixelScale());
194  mRasterizer.clip_box(clip.L, clip.T, clip.R, clip.B);
195 
196  // Add path

198  mRasterizer.add_path(path);
