349  static IColor GetFromHSLA(float h, float s, float l, float a = 1.)
350  {
351  auto hue = [](float h, float m1, float m2)
352  {
353  if (h < 0) h += 1;
354  if (h > 1) h -= 1;
355  if (h < 1.0f / 6.0f)
356  return m1 + (m2 - m1) * h * 6.0f;
357  else if (h < 3.0f / 6.0f)
358  return m2;
359  else if (h < 4.0f / 6.0f)
360  return m1 + (m2 - m1) * (2.0f / 3.0f - h) * 6.0f;
361  return m1;
362  };
363 
364  IColor col;
365  h = std::fmodf(h, 1.0f);
366  if (h < 0.0f) h += 1.0f;
367  s = Clip(s, 0.0f, 1.0f);
368  l = Clip(l, 0.0f, 1.0f);
369  float m2 = l <= 0.5f ? (l * (1 + s)) : (l + s - l * s);
370  float m1 = 2 * l - m2;
371  col.R = static_cast<int>(Clip(hue(h + 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
372  col.G = static_cast<int>(Clip(hue(h, m1, m2), 0.0f, 1.0f) * 255.f);
373  col.B = static_cast<int>(Clip(hue(h - 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
374  col.A = static_cast<int>(a * 255.f);
375  return col;
376  }
