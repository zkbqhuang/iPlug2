156 tresult PLUGIN_API IPlugVST3Controller::notify(IMessage* message)
157 {
158  if (!message)
159  return kInvalidArgument;
160 
161  if (!strcmp(message->getMessageID(), "SCVFD"))
162  {
163  Steinberg::int64 controlTag = kNoTag;
164  double normalizedValue = 0.;
165 
166  if(message->getAttributes()->getInt("CT", controlTag) == kResultFalse)
167  return kResultFalse;
168 
169  if(message->getAttributes()->getFloat("NV", normalizedValue) == kResultFalse)
170  return kResultFalse;
171 
172  SendControlValueFromDelegate((int) controlTag, normalizedValue);
173 
174  }
175  else if (!strcmp(message->getMessageID(), "SCMFD"))
176  {
177  const void* data;
178  Steinberg::int64 controlTag = kNoTag;
179  Steinberg::int64 messageTag = kNoTag;
180 
181  if(message->getAttributes()->getInt("CT", controlTag) == kResultFalse)
182  return kResultFalse;
183 
184  if(message->getAttributes()->getInt("MT", messageTag) == kResultFalse)
185  return kResultFalse;
186 
187  Steinberg::uint32 size;
188 
189  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
190  {
191  SendControlMsgFromDelegate((int) controlTag, (int) messageTag, size, data);
192  return kResultOk;
193  }
194  }
195  else if (!strcmp(message->getMessageID(), "SMMFD"))
196  {
197  const void* data = nullptr;
198  uint32 size;
199 
200  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
201  {
202  if (size == sizeof(IMidiMsg))
203  {
204  IMidiMsg msg;
205  memcpy(&msg, data, size);
206  SendMidiMsgFromDelegate(msg);
207  }
208  }
209  }
210  else if (!strcmp(message->getMessageID(), "SSMFD"))
211  {
212  const void* data = nullptr;
213  uint32 size;
214  int64 offset;
215 
216  if (message->getAttributes()->getBinary("D", data, size) == kResultOk)
217  {
218  if (message->getAttributes()->getInt("O", offset) == kResultOk)
219  {
220  ISysEx msg {static_cast<int>(offset), static_cast<const uint8_t*>(data), static_cast<int>(size)};
221  SendSysexMsgFromDelegate(msg);
222  }
223  }
224  }
225 
226  return ComponentBase::notify(message);
227 }

254 void IPlugVST3Controller::SendArbitraryMsgFromUI(int messageTag, int controlTag, int dataSize, const void* pData)
255 {
256  OPtr<IMessage> message = allocateMessage();
257 
258  if (!message)
259  return;
260 
261  if (dataSize == 0) // allow sending messages with no data
262  {
263  dataSize = 1;
264  uint8_t dummy = 0;
265  pData = &dummy;
266  }
267 
268  message->setMessageID("SAMFUI");
269  message->getAttributes()->setInt("MT", messageTag);
270  message->getAttributes()->setInt("CT", controlTag);
271  message->getAttributes()->setBinary("D", pData, dataSize);
272  sendMessage(message);
273 }
