57 // Utilities for pre-multiplied blits (LICE assumes sources are not pre-multiplied)
58 inline void PreMulCompositeSourceOver(LICE_pixel_chan* out, LICE_pixel_chan* in)
59 {
60  unsigned int alphaCmp = 256 - in[LICE_PIXEL_A];
61 
62  unsigned int A = in[LICE_PIXEL_A] + ((out[LICE_PIXEL_A] * alphaCmp) >> 8);
63  unsigned int R = in[LICE_PIXEL_R] + ((out[LICE_PIXEL_R] * alphaCmp) >> 8);
64  unsigned int G = in[LICE_PIXEL_G] + ((out[LICE_PIXEL_G] * alphaCmp) >> 8);
65  unsigned int B = in[LICE_PIXEL_B] + ((out[LICE_PIXEL_B] * alphaCmp) >> 8);
66 
67  _LICE_MakePixelClamp(out, R, G, B, A);
68 }

70 inline void PreMulCompositeAdd(LICE_pixel_chan* out, LICE_pixel_chan* in)
71 {
72  unsigned int alpha = in[LICE_PIXEL_A];
73 
74  unsigned int A = out[LICE_PIXEL_A] + (255 * in[LICE_PIXEL_A] / alpha);
75  unsigned int R = out[LICE_PIXEL_R] + (255 * in[LICE_PIXEL_R] / alpha);
76  unsigned int G = out[LICE_PIXEL_G] + (255 * in[LICE_PIXEL_G] / alpha);
77  unsigned int B = out[LICE_PIXEL_B] + (255 * in[LICE_PIXEL_B] / alpha);
78 
79  _LICE_MakePixelClamp(out, R, G, B, A);
80 }

82 void PreMulBlit(LICE_IBitmap *dest, LICE_IBitmap *src, int dstx, int dsty, int srcx, int srcy, int srcw, int srch, float alpha, int mode)
83 {
84  srcx = dstx < 0 ? srcx - dstx : srcx;
85  srcy = dsty < 0 ? srcy - dsty : srcy;
86  dstx = std::max(dstx, 0);
87  dsty = std::max(dsty, 0);
88  srcw = std::min(srcw, dest->getWidth() - dstx);
89  srch = std::min(srch, dest->getHeight() - dsty);
90 
91  int inStride = src->getRowSpan() * 4;
92  int outStride = dest->getRowSpan() * 4;
93 
94  LICE_pixel_chan* in = ((LICE_pixel_chan*) src->getBits()) + (srcy * inStride) + (srcx * 4);
95  LICE_pixel_chan* out = ((LICE_pixel_chan*) dest->getBits()) + (dsty * outStride) + (dstx * 4);
96 
97  if ((mode & LICE_BLIT_MODE_MASK) == LICE_BLIT_MODE_ADD)
98  {
99  for (int i = 0; i < srch; i++, in += inStride, out += outStride)
100  {
101  for (int j = 0; j < srcw; j++)
102  PreMulCompositeAdd(out + j * 4, in + j * 4);
103  }
104  }
105  else
106  {
107  for (int i = 0; i < srch; i++, in += inStride, out += outStride)
108  {
109  for (int j = 0; j < srcw; j++)
110  PreMulCompositeSourceOver(out + j * 4, in + j * 4);
111  }
112  }
113 }

425 void IGraphicsLice::FillArc(const IColor& color, float cx, float cy, float r, float aMin, float aMax, const IBlend* pBlend)
426 {
427  NeedsClipping();
428 
429  if (aMax < aMin)
430  std::swap(aMin, aMax);
431 
432  if (aMax >= aMin + 360.f)
433  {
434  FillCircle(color, cx, cy, r, pBlend);
435  return;
436  }
437 
438  float xarray[181];
439  float yarray[181];
440 
441  if (aMax > aMin + 180.f)
442  {
443  if (!OpacityCheck(color, pBlend))
444  {
445  OpacityLayer(&IGraphicsLice::FillArc, pBlend, color, cx, cy, r, aMin, aMax, nullptr);
446  return;
447  }
448 
449  FillArc(color, cx, cy, r, aMin + 178.f, aMax, pBlend);
450  aMax = aMin + 180.f;
451  }
452 
453  aMin = DegToRad(aMin-90.f);
454  aMax = DegToRad(aMax-90.f);
455 
456  int arcpoints = 180.0 * std::min(1., (aMax - aMin) / PI);
457  double arcincrement = (aMax - aMin) / arcpoints;
458  for(int i = 0; i < arcpoints; i++)
459  {
460  xarray[i] = cx + cosf(i * arcincrement + aMin) * r;
461  yarray[i] = cy + sinf(i * arcincrement + aMin) * r;
462  }
463 
464  xarray[arcpoints] = cx;
465  yarray[arcpoints] = cy;
466 
467  FillConvexPolygon(color, xarray, yarray, arcpoints + 1, pBlend);
468 }

482 {

787 void IGraphicsLice::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
788 {
789  const APIBitmap* pBitmap = layer->GetAPIBitmap();
790  LICE_IBitmap* pLayerBitmap = pBitmap->GetBitmap();
791 
792  int stride = pLayerBitmap->getRowSpan() * 4;
793  int size = pLayerBitmap->getHeight() * stride;
794 
795  if (mask.GetSize() >= size)
796  {
797  int x = std::round(shadow.mXOffset * GetScreenScale());
798  int y = std::round(shadow.mYOffset * GetScreenScale());
799  int nRows = pBitmap->GetHeight() - std::abs(y);
800  int nCols = pBitmap->GetWidth() - std::abs(x);
801  LICE_pixel_chan* in = mask.Get() + (std::max(-x, 0) * 4) + (std::max(-y, 0) * stride);
802  LICE_pixel_chan* out = ((LICE_pixel_chan*) pLayerBitmap->getBits()) + (std::max(x, 0) * 4) + (std::max(y, 0) * stride);
803 
804  // Pre-multiply color components
805  IColor color = shadow.mPattern.GetStop(0).mColor;
806  color.Clamp();
807  unsigned int ia = (color.A * static_cast<int>(Clip(shadow.mOpacity, 0.f, 1.f) * 255.0));
808  unsigned int ir = (color.R * ia);
809  unsigned int ig = (color.G * ia);
810  unsigned int ib = (color.B * ia);
811 
812  if (!shadow.mDrawForeground)
813  {
814  LICE_Clear(pLayerBitmap, 0);
815 
816  for (int i = 0; i < nRows; i++, in += stride, out += stride)
817  {
818  LICE_pixel_chan* chans = out;
819 
820  for (int j = 0; j < nCols; j++, chans += 4)
821  {
822  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
823 
824  unsigned int A = (ia * maskAlpha) >> 16;
825  unsigned int R = (ir * maskAlpha) >> 16;
826  unsigned int G = (ig * maskAlpha) >> 16;
827  unsigned int B = (ib * maskAlpha) >> 16;
828 
829  _LICE_MakePixelNoClamp(chans, R, G, B, A);
830  }
831  }
832  }
833  else
834  {
835  for (int i = 0; i < nRows; i++, in += stride, out += stride)
836  {
837  LICE_pixel_chan* chans = out;
838 
839  for (int j = 0; j < nCols; j++, chans += 4)
840  {
841  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
842  unsigned int alphaCmp = 255 - chans[LICE_PIXEL_A];
843 
844  unsigned int A = chans[LICE_PIXEL_A] + ((alphaCmp * ia * maskAlpha) >> 24);
845  unsigned int R = chans[LICE_PIXEL_R] + ((alphaCmp * ir * maskAlpha) >> 24);
846  unsigned int G = chans[LICE_PIXEL_G] + ((alphaCmp * ig * maskAlpha) >> 24);
847  unsigned int B = chans[LICE_PIXEL_B] + ((alphaCmp * ib * maskAlpha) >> 24);
848 
849  _LICE_MakePixelClamp(chans, R, G, B, A);
850  }
851  }
852  }
853  }
