169 tresult PLUGIN_API IPlugVST3::getUnitInfo(int32 unitIndex, UnitInfo& info)
170 {
171  TRACE;
172 
173  if (unitIndex == 0)
174  {
175  info.id = kRootUnitId;
176  info.parentUnitId = kNoParentUnitId;
177  UString name(info.name, 128);
178  name.fromAscii("Root Unit");
179 #ifdef VST3_PRESET_LIST
180  info.programListId = kPresetParam;
181 #else
182  info.programListId = kNoProgramListId;
183 #endif
184  return kResultTrue;
185  }
186  else if (unitIndex > 0 && NParamGroups())
187  {
188  info.id = unitIndex;
189  info.parentUnitId = kRootUnitId;
190  info.programListId = kNoProgramListId;
191 
192  UString name(info.name, 128);
193  name.fromAscii(GetParamGroupName(unitIndex-1));
194 
195  return kResultTrue;
196  }
197 
198  return kResultFalse;
199 }
