204 {
205 public:
206  IVXYPadControl(IRECT bounds, const std::initializer_list<int>& params,
207  const IVColorSpec& colorSpec = DEFAULT_SPEC,
208  float handleRadius = 10.f)
209  : IControl(bounds, params)
210  , IVectorBase(colorSpec)
211  , mHandleRadius(handleRadius)
212  {
213  AttachIControl(this);
214  }
215 
216  void Draw(IGraphics& g) override
217  {
218  float xpos = GetValue(0) * mRECT.W();
219  float ypos = GetValue(1) * mRECT.H();
220 
221  g.DrawVerticalLine(GetColor(kFG), mRECT, 0.5);
222  g.DrawHorizontalLine(GetColor(kFG), mRECT, 0.5);
223  g.FillCircle(GetMouseIsOver() ? GetColor(kHL) : GetColor(kPR), mRECT.L + xpos, mRECT.B - ypos, mHandleRadius);
224  }
225 
226  void OnMouseDown(float x, float y, const IMouseMod& mod) override
227  {
228  OnMouseDrag(x, y, 0., 0., mod);
229  }
230 
231  void OnMouseDrag(float x, float y, float dX, float dY, const IMouseMod& mod) override
232  {
233  mRECT.Constrain(x, y);
234  float xn = (x - mRECT.L) / mRECT.W();
235  float yn = 1.f - ((y - mRECT.T) / mRECT.H());
236  SetValue(xn, 0);
237  SetValue(yn, 1);
238  SetDirty(true);
239  }
