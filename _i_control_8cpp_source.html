622 void IDirBrowseControlBase::ScanDirectory(const char* path, IPopupMenu& menuToAddTo)
623 {
624 #if !defined OS_IOS
625  WDL_DirScan d;
626  IPopupMenu& parentDirMenu = menuToAddTo;
627 
628  if (!d.First(path))
629  {
630  do
631  {
632  const char* f = d.GetCurrentFN();
633  if (f && f[0] != '.')
634  {
635  if (d.GetCurrentIsDirectory())
636  {
637  WDL_String subdir;
638  d.GetCurrentFullFN(&subdir);
639  IPopupMenu* pNewMenu = new IPopupMenu();
640  parentDirMenu.AddItem(d.GetCurrentFN(), pNewMenu, -2);
641  ScanDirectory(subdir.Get(), *pNewMenu);
642  }
643  else
644  {
645  const char* a = strstr(f, mExtension.Get());
646  if (a && a > f && strlen(a) == strlen(mExtension.Get()))
647  {
648  WDL_String menuEntry {f};
649 
650  if(!mShowFileExtensions)
651  menuEntry.Set(f, (int) (a - f));
652 
653  IPopupMenu::Item* pItem = new IPopupMenu::Item(menuEntry.Get(), IPopupMenu::Item::kNoFlags, mFiles.GetSize());
654  parentDirMenu.AddItem(pItem, -2 /* sort alphabetically */);
655  WDL_String* pFullPath = new WDL_String("");
656  d.GetCurrentFullFN(pFullPath);
657  mFiles.Add(pFullPath);
658  }
659  }
660  }
661  } while (!d.Next());
662 
663  menuToAddTo = parentDirMenu;
664  }
665 
666  if(!mShowEmptySubmenus)
667  parentDirMenu.RemoveEmptySubmenus();
668 
669 #endif
670 }
