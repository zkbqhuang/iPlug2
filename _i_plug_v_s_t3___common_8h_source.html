52  template <class T>
53  static bool SetState(T* pPlug, IBStream* pState)
54  {
55  TRACE;
56 
57  IByteChunk chunk;
58 
59  const int bytesPerBlock = 128;
60  char buffer[bytesPerBlock];
61 
62  while(true)
63  {
64  Steinberg::int32 bytesRead = 0;
65  auto status = pState->read(buffer, (Steinberg::int32) bytesPerBlock, &bytesRead);
66 
67  if (bytesRead <= 0 || (status != kResultTrue && pPlug->GetHost() != kHostWaveLab))
68  break;
69 
70  chunk.PutBytes(buffer, bytesRead);
71  }
72  int pos = pPlug->UnserializeState(chunk,0);
73 
74  int32 savedBypass = 0;
75 
76  pState->seek(pos,IBStream::IStreamSeekMode::kIBSeekSet);
77  if (pState->read (&savedBypass, sizeof (Steinberg::int32)) != kResultOk) {
78  return kResultFalse;
79  }
80 
81  IPlugVST3ControllerBase* pController = dynamic_cast<IPlugVST3ControllerBase*>(pPlug);
82 
83  if(pController)
84  {
85  if (pController->mBypassParameter)
86  pController->mBypassParameter->setNormalized(savedBypass);
87  }
88 
89  pPlug->OnRestoreState();
90  return kResultOk;
91  }
