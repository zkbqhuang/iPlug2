141 inline void NanoVGSetBlendMode(NVGcontext* context, const IBlend* pBlend)
142 {
143  if (!pBlend)
144  {
145  nvgGlobalCompositeOperation(context, NVG_SOURCE_OVER);
146  return;
147  }
148 
149  switch (pBlend->mMethod)
150  {
151  case kBlendDefault: // fall through
152  case kBlendClobber: // fall through
153  case kBlendSourceOver: nvgGlobalCompositeOperation(context, NVG_SOURCE_OVER); break;
154  case kBlendSourceIn: nvgGlobalCompositeOperation(context, NVG_SOURCE_IN); break;
155  case kBlendSourceOut: nvgGlobalCompositeOperation(context, NVG_SOURCE_OUT); break;
156  case kBlendSourceAtop: nvgGlobalCompositeOperation(context, NVG_ATOP); break;
157  case kBlendDestOver: nvgGlobalCompositeOperation(context, NVG_DESTINATION_OVER); break;
158  case kBlendDestIn: nvgGlobalCompositeOperation(context, NVG_DESTINATION_IN); break;
159  case kBlendDestOut: nvgGlobalCompositeOperation(context, NVG_DESTINATION_OUT); break;
160  case kBlendDestAtop: nvgGlobalCompositeOperation(context, NVG_DESTINATION_ATOP); break;
161  case kBlendAdd: nvgGlobalCompositeBlendFunc(context, NVG_SRC_ALPHA, NVG_DST_ALPHA); break;
162  case kBlendXOR: nvgGlobalCompositeOperation(context, NVG_XOR); break;
163  }
164 }

166 NVGpaint NanoVGPaint(NVGcontext* pContext, const IPattern& pattern, const IBlend* pBlend)
167 {
168  double s[2], e[2];
169 
170  NVGcolor icol = NanoVGColor(pattern.GetStop(0).mColor, pBlend);
171  NVGcolor ocol = NanoVGColor(pattern.GetStop(pattern.NStops() - 1).mColor, pBlend);
172 
173  // Invert transform
174  IMatrix inverse = IMatrix(pattern.mTransform).Invert();
175  inverse.TransformPoint(s[0], s[1], 0.0, 0.0);
176 
177  if (pattern.mType == kRadialPattern)
178  {
179  return nvgRadialGradient(pContext, s[0], s[1], 0.0, inverse.mXX, icol, ocol);
180  }
181  else
182  {
183  inverse.TransformPoint(e[0], e[1], 0.0, 1.0);
184 
185  return nvgLinearGradient(pContext, s[0], s[1], e[0], e[1], icol, ocol);
186  }
187 }

234 IBitmap IGraphicsNanoVG::LoadBitmap(const char* name, int nStates, bool framesAreHorizontal, int targetScale)
235 {
236  if (targetScale == 0)
237  targetScale = GetScreenScale();
238 
239  // NanoVG does not use the global static cache, since bitmaps are textures linked to a context
240  StaticStorage<APIBitmap>::Accessor storage(mBitmapCache);
241  APIBitmap* pAPIBitmap = storage.Find(name, targetScale);
242 
243  // If the bitmap is not already cached at the targetScale
244  if (!pAPIBitmap)
245  {
246  const char* ext = name + strlen(name) - 1;
247  while (ext >= name && *ext != '.') --ext;
248  ++ext;
249 
250  WDL_String fullPathOrResourceID;
251  int sourceScale = 0;
252  EResourceLocation resourceFound = SearchImageResource(name, ext, fullPathOrResourceID, targetScale, sourceScale);
253 
254  bool bitmapTypeSupported = BitmapExtSupported(ext);
255 
256  if(resourceFound == EResourceLocation::kNotFound || !bitmapTypeSupported)
257  {
258  assert("Bitmap not found");
259  return IBitmap(); // return invalid IBitmap
260  }
261 
262  pAPIBitmap = LoadAPIBitmap(fullPathOrResourceID.Get(), sourceScale, resourceFound, ext);
263 
264  storage.Add(pAPIBitmap, name, sourceScale);
265 
266  assert(pAPIBitmap && "Bitmap not loaded");
267  }
268 
269  return IBitmap(pAPIBitmap, nStates, framesAreHorizontal, name);
270 }

330 void IGraphicsNanoVG::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
331 {
332  const APIBitmap* pBitmap = layer->GetAPIBitmap();
333  int width = pBitmap->GetWidth();
334  int height = pBitmap->GetHeight();
335  int size = width * height * 4;
336 
337  if (mask.GetSize() >= size)
338  {
339  if (!shadow.mDrawForeground)
340  {
341  PushLayer(layer.get());
342  nvgGlobalCompositeBlendFunc(mVG, NVG_ZERO, NVG_ZERO);
343  PathRect(layer->Bounds());
344  nvgFillColor(mVG, NanoVGColor(COLOR_TRANSPARENT));
345  nvgFill(mVG);
346  PopLayer();
347  }
348 
349  IRECT bounds(layer->Bounds());
350 
351  NanoVGBitmap maskRawBitmap(mVG, width, height, mask.Get(), pBitmap->GetScale(), pBitmap->GetDrawScale());
352  APIBitmap* shadowBitmap = CreateAPIBitmap(width, height, pBitmap->GetScale(), pBitmap->GetDrawScale());
353  IBitmap tempLayerBitmap(shadowBitmap, 1, false);
354  IBitmap maskBitmap(&maskRawBitmap, 1, false);
355  ILayer shadowLayer(shadowBitmap, layer->Bounds());
356 
357  PathTransformSave();
358  PushLayer(layer.get());
359  PushLayer(&shadowLayer);
360  DrawBitmap(maskBitmap, bounds, 0, 0, nullptr);
361  IBlend blend1(kBlendSourceIn, 1.0);
362  PathRect(layer->Bounds());
363  PathTransformTranslate(-shadow.mXOffset, -shadow.mYOffset);
364  PathFill(shadow.mPattern, IFillOptions(), &blend1);
365  PopLayer();
366  IBlend blend2(kBlendDestOver, shadow.mOpacity);
367  bounds.Translate(shadow.mXOffset, shadow.mYOffset);
368  DrawBitmap(tempLayerBitmap, bounds, 0, 0, &blend2);
369  PopLayer();
370  PathTransformRestore();
371  }
372 }

467 void IGraphicsNanoVG::DrawBitmap(const IBitmap& bitmap, const IRECT& dest, int srcX, int srcY, const IBlend* pBlend)
468 {
469  APIBitmap* pAPIBitmap = bitmap.GetAPIBitmap();
470 
471  assert(pAPIBitmap);
472 
473  // First generate a scaled image paint
474  NVGpaint imgPaint;
475  double scale = 1.0 / (pAPIBitmap->GetScale() * pAPIBitmap->GetDrawScale());
476 
477  nvgTransformScale(imgPaint.xform, scale, scale);
478 
479  imgPaint.xform[4] = dest.L - srcX;
480  imgPaint.xform[5] = dest.T - srcY;
481  imgPaint.extent[0] = bitmap.W() * bitmap.GetScale();
482  imgPaint.extent[1] = bitmap.H() * bitmap.GetScale();
483  imgPaint.image = pAPIBitmap->GetBitmap();
484  imgPaint.radius = imgPaint.feather = 0.f;
485  imgPaint.innerColor = imgPaint.outerColor = nvgRGBAf(1, 1, 1, BlendWeight(pBlend));
486 
487  // Now draw
488 
489  nvgBeginPath(mVG); // Clears any existing path
490  nvgRect(mVG, dest.L, dest.T, dest.W(), dest.H());
491  nvgFillPaint(mVG, imgPaint);
492  NanoVGSetBlendMode(mVG, pBlend);
493  nvgFill(mVG);
494  nvgGlobalCompositeOperation(mVG, NVG_SOURCE_OVER);
495  nvgBeginPath(mVG); // Clears the bitmap rect from the path state
496 }

533 void IGraphicsNanoVG::PrepareAndMeasureText(const IText& text, const char* str, IRECT& r, double& x, double & y) const
534 {
535  float fbounds[4];
536 
537  assert(nvgFindFont(mVG, text.mFont) != -1 && "No font found - did you forget to load it?");
538 
539  nvgFontBlur(mVG, 0);
540  nvgFontSize(mVG, text.mSize);
541  nvgFontFace(mVG, text.mFont);
542 
543  int align = 0;
544 
545  switch (text.mAlign)
546  {
547  case IText::kAlignNear: align = NVG_ALIGN_LEFT; x = r.L; break;
548  case IText::kAlignCenter: align = NVG_ALIGN_CENTER; x = r.MW(); break;
549  case IText::kAlignFar: align = NVG_ALIGN_RIGHT; x = r.R; break;
550  }
551 
552  switch (text.mVAlign)
553  {
554  case IText::kVAlignTop: align |= NVG_ALIGN_TOP; y = r.T; break;
555  case IText::kVAlignMiddle: align |= NVG_ALIGN_MIDDLE; y = r.MH(); break;
556  case IText::kVAlignBottom: align |= NVG_ALIGN_BOTTOM; y = r.B; break;
557  }
558 
559  nvgTextAlign(mVG, align);
560  nvgTextBounds(mVG, x, y, str, NULL, fbounds);
561 
562  r = IRECT(fbounds[0], fbounds[1], fbounds[2], fbounds[3]);
563 }

588 void IGraphicsNanoVG::PathStroke(const IPattern& pattern, float thickness, const IStrokeOptions& options, const IBlend* pBlend)
589 {
590  // First set options
591  switch (options.mCapOption)
592  {
593  case kCapButt: nvgLineCap(mVG, NVG_BUTT); break;
594  case kCapRound: nvgLineCap(mVG, NVG_ROUND); break;
595  case kCapSquare: nvgLineCap(mVG, NVG_SQUARE); break;
596  }
597 
598  switch (options.mJoinOption)
599  {
600  case kJoinMiter: nvgLineJoin(mVG, NVG_MITER); break;
601  case kJoinRound: nvgLineJoin(mVG, NVG_ROUND); break;
602  case kJoinBevel: nvgLineJoin(mVG, NVG_BEVEL); break;
603  }
604 
605  nvgMiterLimit(mVG, options.mMiterLimit);

712 void IGraphicsNanoVG::DrawDottedLine(const IColor& color, float x1, float y1, float x2, float y2, const IBlend* pBlend, float thickness, float dashLen)
713 {
714  const float xd = x1 - x2;
715  const float yd = y1 - y2;
716  const float len = std::sqrt(xd * xd + yd * yd);
717 
718  const float segs = std::round(len / dashLen);
719  const float incr = 1.f / segs;
720 
721  float xs = x1;
722  float ys = y1;
723 
724  PathMoveTo(xs, ys);
725 
726  for (int i = 1; i < static_cast<int>(segs); i+=2)
727  {
728  float progress = incr * static_cast<float>(i);
729 
730  float xe = x1 + progress * (x2 - x1);
731  float ye = y1 + progress * (y2 - y1);
732 
733  PathLineTo(xe, ye);
734 
735  progress += incr;
736 
737  xs = x1 + progress * (x2 - x1);
738  ys = y1 + progress * (y2 - y1);
739 
740  PathMoveTo(xs, ys);
741  }
742 
743  PathStroke(color, thickness, IStrokeOptions(), pBlend);
744 }

746 void IGraphicsNanoVG::DrawDottedRect(const IColor& color, const IRECT& bounds, const IBlend* pBlend, float thickness, float dashLen)
747 {
748  const int xsegs = static_cast<int>(std::ceil(bounds.W() / (dashLen * 2.f)));
749  const int ysegs = static_cast<int>(std::ceil(bounds.H() / (dashLen * 2.f)));
750 
751  float x1 = bounds.L;
752  float y1 = bounds.T;
753 
754  float x2 = x1;
755  float y2 = y1;
756 
757  PathMoveTo(x1, y1);
758 
759  for(int j = 0; j < 2; j++)
760  {
761  for (int i = 0; i < xsegs; i++)
762  {
763  x2 = Clip(x1 + dashLen, bounds.L, bounds.R);
764  PathLineTo(x2, y2);
765  x1 = Clip(x2 + dashLen, bounds.L, bounds.R);
766  PathMoveTo(x1, y1);
767  }
768 
769  x2 = x1;
770 
771  for (int i = 0; i < ysegs; i++)
772  {
773  y2 = Clip(y1 + dashLen, bounds.T, bounds.B);
774  PathLineTo(x2, y2);
775  y1 = Clip(y2 + dashLen, bounds.T, bounds.B);
776  PathMoveTo(x1, y1);
777  }
778 
779  y2 = y1;
780 
781  dashLen = -dashLen;
782  }
783 
784  PathStroke(color, thickness, IStrokeOptions(), pBlend);
785 }
