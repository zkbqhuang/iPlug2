23 class IPlugVST3Parameter : public Parameter

26  IPlugVST3Parameter(IParam* pParam, ParamID tag, UnitID unitID)
27  : mIPlugParam(pParam)
28  {
29  UString(info.title, str16BufferSize(String128)).assign(pParam->GetNameForHost());
30  UString(info.units, str16BufferSize(String128)).assign(pParam->GetLabelForHost());
31 
32  precision = pParam->GetDisplayPrecision();
33 
34  if (pParam->Type() != IParam::kTypeDouble)
35  info.stepCount = pParam->GetRange();
36  else
37  info.stepCount = 0; // continuous
38 
39  int32 flags = 0;
40 
41  if (pParam->GetCanAutomate()) flags |= ParameterInfo::kCanAutomate;
42  if (pParam->Type() == IParam::kTypeEnum) flags |= ParameterInfo::kIsList;
43 
44  info.defaultNormalizedValue = valueNormalized = pParam->ToNormalized(pParam->GetDefault());
45  info.flags = flags;
46  info.id = tag;
47  info.unitId = unitID;
48  }
