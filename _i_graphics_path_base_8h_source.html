438  IPattern GetSVGPattern(const NSVGpaint& paint, float opacity)
439  {
440  int alpha = std::min(255, std::max(int(roundf(opacity * 255.f)), 0));
441 
442  switch (paint.type)
443  {
444  case NSVG_PAINT_COLOR:
445  return IColor(alpha, (paint.color >> 0) & 0xFF, (paint.color >> 8) & 0xFF, (paint.color >> 16) & 0xFF);
446 
447  case NSVG_PAINT_LINEAR_GRADIENT:
448  case NSVG_PAINT_RADIAL_GRADIENT:
449  {
450  NSVGgradient* pGrad = paint.gradient;
451 
452  IPattern pattern(paint.type == NSVG_PAINT_LINEAR_GRADIENT ? kLinearPattern : kRadialPattern);
453 
454  // Set Extend Rule
455  switch (pGrad->spread)
456  {
457  case NSVG_SPREAD_PAD: pattern.mExtend = kExtendPad; break;
458  case NSVG_SPREAD_REFLECT: pattern.mExtend = kExtendReflect; break;
459  case NSVG_SPREAD_REPEAT: pattern.mExtend = kExtendRepeat; break;
460  }
461 
462  // Copy Stops
463  for (int i = 0; i < pGrad->nstops; i++)
464  {
465  unsigned int color = pGrad->stops[i].color;
466  pattern.AddStop(IColor(255, (color >> 0) & 0xFF, (color >> 8) & 0xFF, (color >> 16) & 0xFF), pGrad->stops[i].offset);
467  }
468 
469  // Copy transform
470  pattern.SetTransform(pGrad->xform[0], pGrad->xform[1], pGrad->xform[2], pGrad->xform[3], pGrad->xform[4], pGrad->xform[5]);
471 
472  return pattern;
473  }
474  default:
475  return IColor(alpha, 0, 0, 0);
476  }

479  void RenderNanoSVG(NSVGimage* pImage)
480  {
481  assert(pImage != nullptr);
482 
483  for (NSVGshape* pShape = pImage->shapes; pShape; pShape = pShape->next)
484  {
485  if (!(pShape->flags & NSVG_FLAGS_VISIBLE))
486  continue;
487 
488  PathClear();
489 
490  for (NSVGpath* pPath = pShape->paths; pPath; pPath = pPath->next)
491  {
492  PathMoveTo(pPath->pts[0], pPath->pts[1]);
493 
494  for (int i = 1; i < pPath->npts; i += 3)
495  {
496  float *p = pPath->pts + i * 2;
497  PathCurveTo(p[0], p[1], p[2], p[3], p[4], p[5]);
498  }
499 
500  if (pPath->closed)
501  PathClose();
502  }
503 
504  // Fill
505  if (pShape->fill.type != NSVG_PAINT_NONE)
506  {
507  IFillOptions options;
508 
509  if (pShape->fillRule == NSVG_FILLRULE_EVENODD)
510  options.mFillRule = kFillEvenOdd;
511  else
512  options.mFillRule = kFillWinding;
513 
514  options.mPreserve = pShape->stroke.type != NSVG_PAINT_NONE;
515  PathFill(GetSVGPattern(pShape->fill, pShape->opacity), options, nullptr);
516  }
517 
518  // Stroke
519  if (pShape->stroke.type != NSVG_PAINT_NONE)
520  {
521  IStrokeOptions options;
522 
523  options.mMiterLimit = pShape->miterLimit;
524 
525  switch (pShape->strokeLineCap)
526  {
527  case NSVG_CAP_BUTT: options.mCapOption = kCapButt; break;
528  case NSVG_CAP_ROUND: options.mCapOption = kCapRound; break;
529  case NSVG_CAP_SQUARE: options.mCapOption = kCapSquare; break;
530  }
531 
532  switch (pShape->strokeLineJoin)
533  {
534  case NSVG_JOIN_MITER: options.mJoinOption = kJoinMiter; break;
535  case NSVG_JOIN_ROUND: options.mJoinOption = kJoinRound; break;
536  case NSVG_JOIN_BEVEL: options.mJoinOption = kJoinBevel; break;
537  }
538 
539  options.mDash.SetDash(pShape->strokeDashArray, pShape->strokeDashOffset, pShape->strokeDashCount);
540 
541  PathStroke(GetSVGPattern(pShape->stroke, pShape->opacity), pShape->strokeWidth, options, nullptr);
542  }
543  }
