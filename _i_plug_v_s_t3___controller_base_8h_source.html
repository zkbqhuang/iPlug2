136  void PLUGIN_API setParamNormalized(IPlugAPIBase* pPlug, ParamID tag, ParamValue value)
137  {
138  if (tag >= kBypassParam)
139  {
140  switch (tag)
141  {
142  case kPresetParam:

144  pPlug->RestorePreset(pPlug->NPresets() * value);
145  break;
146  }
147  default:
148  break;
149  }
150  }
151  else
152  {
153  IParam* pParam = pPlug->GetParam(tag);
154 
155  if (pParam)
156  {
157  pParam->SetNormalized(value);
158  pPlug->OnParamChangeUI(tag, kHost);
159  pPlug->SendParameterValueFromAPI(tag, value, true);
160  }
